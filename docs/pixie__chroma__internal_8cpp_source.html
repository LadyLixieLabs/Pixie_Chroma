39  void ICACHE_RAM_ATTR ANIMATE(){
40  extern PixieChroma pix;
41  // This is an imperfect solution, but extern-ing the class instance defined in the
42  // user sketch, this ISR routine (which can't be in a class) can still access it.
43  // _pixie_animations.cpp does this trick as well. This is why you cannot currently
44  // name your PixieChroma class instance anything but "pix". :/
45  //
46  // The solution to this would be the ISR setting a flag that a pix.process() function would read, but
47  // That would mean users can't ever delay() or use other blocking code for more than a split
48  // second without having to call the "pix.process()" again to avoid choppy output.
49  //
50  // At the cost of some non-standard methods of accessing the class in here, beginners to Arduino
51  // can just use simple print() and clear() methods to update the displays, while our
52  // tricky ISR keeps animation running nicely, even with blocking code in the main loop.
53 
54  uint32_t t_now = micros(); // Get current time
55  pix.frame_time = (t_now - pix.t_last);
56  pix.t_last = t_now;
57 
58  anim_func(); // Call custom animation function
59 
60  pix.show(); // Update Pixie Chromas
61 
62  pix.frame_iter += 1;

668 void PixieChroma::write_pix(char* message, int16_t x_pos, int16_t y_pos){
669  uint8_t len = strlen(message);
670  int16_t x_offset = x_pos;
671  int16_t y_offset = y_pos;
672  for(uint8_t i = 0; i < len; i++){
673  if(message[i] == '\n'){
674  y_offset += display_height;
675  x_offset = 1;
676  cursor_x = 1;
677  cursor_y += display_height;
678  }
679  else if(message[i] == 0 || message[i] == '\0'){ // early end of string
680  return;
681  }
682  else{
683  add_char(message[i], cursor_x, cursor_y);
684  x_offset += display_width;
685  cursor_x += display_width;
686  }
687  }
688 }

703 void PixieChroma::add_char(char chr, int16_t x_pos, int16_t y_pos){
704  if (chr >= 32) {
705  chr -= 32;
706  }
707 
708  for(uint8_t x = 0; x < 5; x++){
709  uint8_t column = pgm_read_byte(font+(chr * 5 + x));
710 
711  uint16_t row1_index = xy(x_pos+x, y_pos+0);
712  uint16_t row2_index = xy(x_pos+x, y_pos+1);
713  uint16_t row3_index = xy(x_pos+x, y_pos+2);
714  uint16_t row4_index = xy(x_pos+x, y_pos+3);
715  uint16_t row5_index = xy(x_pos+x, y_pos+4);
716  uint16_t row6_index = xy(x_pos+x, y_pos+5);
717  uint16_t row7_index = xy(x_pos+x, y_pos+6);
718 
719  // "Subtract" from mask transparency with saturating function
720  mask[row1_index] = qadd8( mask[row1_index], bit_table[bitRead(column,0)] );
721  mask[row2_index] = qadd8( mask[row2_index], bit_table[bitRead(column,1)] );
722  mask[row3_index] = qadd8( mask[row3_index], bit_table[bitRead(column,2)] );
723  mask[row4_index] = qadd8( mask[row4_index], bit_table[bitRead(column,3)] );
724  mask[row5_index] = qadd8( mask[row5_index], bit_table[bitRead(column,4)] );
725  mask[row6_index] = qadd8( mask[row6_index], bit_table[bitRead(column,5)] );
726  mask[row7_index] = qadd8( mask[row7_index], bit_table[bitRead(column,6)] );
727  }

1134 /**************************************************************************/
1140 /**************************************************************************/
1141 void PixieChroma::blur_x(fract8 blur_amount){
1142  uint8_t keep = 255 - blur_amount;
1143  uint8_t seep = blur_amount >> 1;
1144  for( uint8_t row = 0; row < matrix_height; row++) {
1145  uint8_t carryover = 0;
1146  for( uint8_t i = 0; i < matrix_width; i++) {
1147  uint8_t cur = mask[xy(i,row)];
1148  uint8_t part = cur;
1149  part = scale8(part, seep);
1150  cur = scale8(cur, keep);
1151  cur = qadd8(cur,carryover);
1152  if(i){
1153  mask[xy(i-1,row)] = qadd8(mask[xy(i-1,row)],part);
1154  }
1155  mask[xy(i,row)] = cur;
1156  carryover = part;
1157  }
1158  }
1159 }

1516 /**************************************************************************/
1525 /**************************************************************************/
1526 void PixieChroma::draw_line(int16_t x1, int16_t y1, int16_t x2, int16_t y2){
1527  //Bresenham's line algorithm
1528  uint16_t index;
1529 
1530  int16_t x,y,dx,dy,dx1,dy1,px,py,xe,ye,i;
1531  dx=x2-x1;
1532  dy=y2-y1;
1533  dx1=fabs(dx);
1534  dy1=fabs(dy);
1535  px=2*dy1-dx1;
1536  py=2*dx1-dy1;
1537  if(dy1<=dx1){
1538  if(dx>=0){
1539  x=x1;
1540  y=y1;
1541  xe=x2;
1542  }
1543  else{
1544  x=x2;
1545  y=y2;
1546  xe=x1;
1547  }
1548  index = xy(x,y);
1549  mask[index] = qadd8( 255, mask[index] );
1550  for(i=0;x<xe;i++){
1551  x=x+1;
1552  if(px<0){
1553  px=px+2*dy1;
1554  }
1555  else{
1556  if((dx<0 && dy<0) || (dx>0 && dy>0)){
1557  y=y+1;
1558  }
1559  else{
1560  y=y-1;
1561  }
1562  px=px+2*(dy1-dx1);
1563  }
1564  delay(0);
1565  index = xy(x,y);
1566  mask[index] = qadd8( 255, mask[index] );
1567  }
1568  }
1569  else{
1570  if(dy>=0){
1571  x=x1;
1572  y=y1;
1573  ye=y2;
1574  }
1575  else{
1576  x=x2;
1577  y=y2;
1578  ye=y1;
1579  }
1580  index = xy(x,y);
1581  mask[index] = qadd8( 255, mask[index] );
1582  for(i=0;y<ye;i++){
1583  y=y+1;
1584  if(py<=0){
1585  py=py+2*dx1;
1586  }
1587  else{
1588  if((dx<0 && dy<0) || (dx>0 && dy>0)){
1589  x=x+1;
1590  }
1591  else{
1592  x=x-1;
1593  }
1594  py=py+2*(dx1-dy1);
1595  }
1596  index = xy(x,y);
1597  mask[index] = qadd8( 255, mask[index] );
1598  }
1599  }

1603 /**************************************************************************/
1621 /**************************************************************************/
1622 CRGB PixieChroma::kelvin_to_rgb(uint16_t temperature){
1623  // Tanner Helland formula
1624  float _temperature;
1625  float _red;
1626  float _green;
1627  float _blue;
1628 
1629  _temperature = constrain(temperature, 0, 65500);
1630 
1631  _red = _green = _blue = 0;
1632  float t = _temperature * 0.01;
1633 
1634  if (t <= 66){
1635  _red = 255;
1636  _green = (99.4708025861 * log(t)) - 161.1195681661;
1637  if (t > 19){
1638  _blue = (138.5177312231 * log(t - 10)) - 305.0447927307;
1639  }
1640  else{
1641  _blue = 0;
1642  }
1643  }
1644  else{
1645  _red = 329.698727466 * pow(t - 60, -0.1332047592);
1646  _green = 288.1221695283 * pow(t - 60, -0.0755148492);
1647  _blue = 255;
1648  }
1649 
1650  float f = 0.01 * 100;
1651 
1652  _red = constrain(f * _red, 0, 255);
1653  _green = constrain(f * _green, 0, 255);
1654  _blue = constrain(f * _blue, 0, 255);
1655 
1656  return CRGB(_red, _green, _blue);
1657 }
