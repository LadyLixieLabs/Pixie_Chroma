576 void PixieChroma::write_pix(char* message, int16_t x_pos, int16_t y_pos){
577  uint8_t len = strlen(message);
578  int16_t x_offset = x_pos;
579  int16_t y_offset = y_pos;
580  for(uint8_t i = 0; i < len; i++){
581  if(message[i] == '\n'){
582  y_offset += display_height;
583  x_offset = 1;
584  cursor_x = 1;
585  cursor_y += display_height;
586  }
587  else if(message[i] == 0 || message[i] == '\0'){ // early end of string
588  return;
589  }
590  else{
591  add_char(message[i], cursor_x, cursor_y);
592  x_offset += display_width;
593  cursor_x += display_width;
594  }
595  }
596 }

1042 /**************************************************************************/
1048 /**************************************************************************/
1049 void PixieChroma::blur_x(fract8 blur_amount){
1050  uint8_t keep = 255 - blur_amount;
1051  uint8_t seep = blur_amount >> 1;
1052  for( uint8_t row = 0; row < matrix_height; row++) {
1053  uint8_t carryover = 0;
1054  for( uint8_t i = 0; i < matrix_width; i++) {
1055  uint8_t cur = mask[xy(i,row)];
1056  uint8_t part = cur;
1057  part = scale8(part, seep);
1058  cur = scale8(cur, keep);
1059  cur = qadd8(cur,carryover);
1060  if(i){
1061  mask[xy(i-1,row)] = qadd8(mask[xy(i-1,row)],part);
1062  }
1063  mask[xy(i,row)] = cur;
1064  carryover = part;
1065  }
1066  }
1067 }

1076 /**************************************************************************/
1077 void PixieChroma::blur_y(fract8 blur_amount){
1078  // blur columns
1079  uint8_t keep = 255 - blur_amount;
1080  uint8_t seep = blur_amount >> 1;
1081  for( uint8_t col = 0; col < matrix_width; ++col) {
1082  uint8_t carryover = 0;
1083  for( uint8_t i = 0; i < matrix_height; ++i) {
1084  uint8_t cur = mask[xy(col,i)];
1085  uint8_t part = cur;
1086  part = scale8(part, seep);
1087  cur = scale8(cur, keep);
1088  cur = qadd8(cur,carryover);
1089  if(i){
1090  mask[xy(col,i-1)] = qadd8(mask[xy(col,i-1)],part);
1091  }
1092  mask[xy(col,i)] = cur;
1093  carryover = part;
1094  }
1095  }

1434 void PixieChroma::draw_line(int16_t x1, int16_t y1, int16_t x2, int16_t y2){

1436  uint16_t index;
1437 
1438  int16_t x,y,dx,dy,dx1,dy1,px,py,xe,ye,i;
1439  dx=x2-x1;
1440  dy=y2-y1;
1441  dx1=fabs(dx);
1442  dy1=fabs(dy);
1443  px=2*dy1-dx1;
1444  py=2*dx1-dy1;
1445  if(dy1<=dx1){
1446  if(dx>=0){
1447  x=x1;
1448  y=y1;
1449  xe=x2;
1450  }
1451  else{
1452  x=x2;
1453  y=y2;
1454  xe=x1;
1455  }
1456  index = xy(x,y);
1457  mask[index] = qadd8( 255, mask[index] );
1458  for(i=0;x<xe;i++){
1459  x=x+1;
1460  if(px<0){
1461  px=px+2*dy1;
1462  }
1463  else{
1464  if((dx<0 && dy<0) || (dx>0 && dy>0)){
1465  y=y+1;
1466  }
1467  else{
1468  y=y-1;
1469  }
1470  px=px+2*(dy1-dx1);
1471  }
1472  delay(0);
1473  index = xy(x,y);
1474  mask[index] = qadd8( 255, mask[index] );
1475  }
1476  }
1477  else{
1478  if(dy>=0){
1479  x=x1;
1480  y=y1;
1481  ye=y2;
1482  }
1483  else{
1484  x=x2;
1485  y=y2;
1486  ye=y1;
1487  }

1492  if(py<=0){
1493  py=py+2*dx1;
1494  }
1495  else{
1496  if((dx<0 && dy<0) || (dx>0 && dy>0)){
1497  x=x+1;
1498  }
1499  else{

1501  }

1530 CRGB PixieChroma::kelvin_to_rgb(uint16_t temperature){
1531  // Tanner Helland formula
1532  float _temperature;
1533  float _red;
1534  float _green;
1535  float _blue;
1536 
1537  _temperature = constrain(temperature, 0, 65500);
1538 
1539  _red = _green = _blue = 0;
1540  float t = _temperature * 0.01;
1541 
1542  if (t <= 66){
1543  _red = 255;
1544  _green = (99.4708025861 * log(t)) - 161.1195681661;
1545  if (t > 19){
1546  _blue = (138.5177312231 * log(t - 10)) - 305.0447927307;
1547  }
1548  else{
1549  _blue = 0;
1550  }
1551  }
1552  else{
1553  _red = 329.698727466 * pow(t - 60, -0.1332047592);
1554  _green = 288.1221695283 * pow(t - 60, -0.0755148492);
1555  _blue = 255;
1556  }
1557 
1558  float f = 0.01 * 100;
1559 
1560  _red = constrain(f * _red, 0, 255);
1561  _green = constrain(f * _green, 0, 255);
1562  _blue = constrain(f * _blue, 0, 255);
1563 
1564  return CRGB(_red, _green, _blue);
