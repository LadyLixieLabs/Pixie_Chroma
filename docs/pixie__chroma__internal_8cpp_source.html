733 /**************************************************************************/
746 /**************************************************************************/
747 void PixieChroma::write_pix(char* message, int16_t x_offset, int16_t y_offset){
748  uint8_t len = strlen(message);
749  for(uint8_t i = 0; i < len; i++){
750  if(message[i] == '\n'){ // Newline, force line break
751  // TODO: write_pix(char*) should never modify cursor position
752  // Use an internal offset (+user offset) to place chars instead
753  cursor_x = 1;
754  cursor_y += display_height;
755  }
756  else if(line_wrap == true && cursor_x >= (display_width*chars_x)){ // End of line reached, wrap to new line if line_wrap enabled
757  cursor_x = 1;
758  cursor_y += display_height;
759  add_char(message[i], cursor_x+x_offset, cursor_y+y_offset);
760  cursor_x += display_width;
761  }
762  else if(message[i] == 0 || message[i] == '\0'){ // end of string
763  return;
764  }
765  else{ // Normal
766  add_char(message[i], cursor_x+x_offset, cursor_y+y_offset);
767  cursor_x += display_width;
768  }
769  }
770 }

785 void PixieChroma::add_char(char chr, int16_t x_pos, int16_t y_pos){
786  if (chr >= 32) {
787  chr -= 32;
788  }
789 
790  for(uint8_t x = 0; x < 5; x++){
791  uint8_t column = pgm_read_byte(font+(chr * 5 + x));
792 
793  uint16_t row1_index = xy(x_pos+x, y_pos+0);
794  uint16_t row2_index = xy(x_pos+x, y_pos+1);
795  uint16_t row3_index = xy(x_pos+x, y_pos+2);
796  uint16_t row4_index = xy(x_pos+x, y_pos+3);
797  uint16_t row5_index = xy(x_pos+x, y_pos+4);
798  uint16_t row6_index = xy(x_pos+x, y_pos+5);
799  uint16_t row7_index = xy(x_pos+x, y_pos+6);
800 
801  // "Subtract" from mask transparency with saturating function
802  mask[row1_index] = qadd8( mask[row1_index], bit_table[bitRead(column,0)] );
803  mask[row2_index] = qadd8( mask[row2_index], bit_table[bitRead(column,1)] );
804  mask[row3_index] = qadd8( mask[row3_index], bit_table[bitRead(column,2)] );
805  mask[row4_index] = qadd8( mask[row4_index], bit_table[bitRead(column,3)] );
806  mask[row5_index] = qadd8( mask[row5_index], bit_table[bitRead(column,4)] );
807  mask[row6_index] = qadd8( mask[row6_index], bit_table[bitRead(column,5)] );
808  mask[row7_index] = qadd8( mask[row7_index], bit_table[bitRead(column,6)] );
809  }

1216 /**************************************************************************/
1222 /**************************************************************************/
1223 void PixieChroma::blur_x(fract8 blur_amount){
1224  uint8_t keep = 255 - blur_amount;
1225  uint8_t seep = blur_amount >> 1;
1226  for( uint8_t row = 0; row < matrix_height; row++) {
1227  uint8_t carryover = 0;
1228  for( uint8_t i = 0; i < matrix_width; i++) {
1229  uint8_t cur = mask[xy(i,row)];
1230  uint8_t part = cur;
1231  part = scale8(part, seep);
1232  cur = scale8(cur, keep);
1233  cur = qadd8(cur,carryover);
1234  if(i){
1235  mask[xy(i-1,row)] = qadd8(mask[xy(i-1,row)],part);
1236  }
1237  mask[xy(i,row)] = cur;
1238  carryover = part;
1239  }
1240  }
1241 }

1244 /**************************************************************************/
1250 /**************************************************************************/
1251 void PixieChroma::blur_y(fract8 blur_amount){
1252  // blur columns
1253  uint8_t keep = 255 - blur_amount;
1254  uint8_t seep = blur_amount >> 1;
1255  for( uint8_t col = 0; col < matrix_width; ++col) {
1256  uint8_t carryover = 0;
1257  for( uint8_t i = 0; i < matrix_height; ++i) {
1258  uint8_t cur = mask[xy(col,i)];
1259  uint8_t part = cur;
1260  part = scale8(part, seep);
1261  cur = scale8(cur, keep);
1262  cur = qadd8(cur,carryover);
1263  if(i){

1584 /**************************************************************************/
1585 void PixieChroma::draw_line(int16_t x1, int16_t y1, int16_t x2, int16_t y2){
1586  //Bresenham's line algorithm
1587  uint16_t index;
1588 
1589  int16_t x,y,dx,dy,dx1,dy1,px,py,xe,ye,i;
1590  dx=x2-x1;
1591  dy=y2-y1;
1592  dx1=fabs(dx);
1593  dy1=fabs(dy);
1594  px=2*dy1-dx1;
1595  py=2*dx1-dy1;
1596  if(dy1<=dx1){
1597  if(dx>=0){
1598  x=x1;
1599  y=y1;
1600  xe=x2;
1601  }
1602  else{
1603  x=x2;
1604  y=y2;
1605  xe=x1;
1606  }
1607  index = xy(x,y);
1608  mask[index] = qadd8( 255, mask[index] );
1609  for(i=0;x<xe;i++){
1610  x=x+1;
1611  if(px<0){
1612  px=px+2*dy1;
1613  }
1614  else{
1615  if((dx<0 && dy<0) || (dx>0 && dy>0)){
1616  y=y+1;
1617  }
1618  else{
1619  y=y-1;
1620  }
1621  px=px+2*(dy1-dx1);
1622  }
1623  delay(0);
1624  index = xy(x,y);
1625  mask[index] = qadd8( 255, mask[index] );
1626  }
1627  }
1628  else{
1629  if(dy>=0){
1630  x=x1;
1631  y=y1;
1632  ye=y2;
1633  }
1634  else{
1635  x=x2;
1636  y=y2;
1637  ye=y1;
1638  }
1639  index = xy(x,y);
1640  mask[index] = qadd8( 255, mask[index] );
1641  for(i=0;y<ye;i++){
1642  y=y+1;
1643  if(py<=0){
1644  py=py+2*dx1;
1645  }
1646  else{
1647  if((dx<0 && dy<0) || (dx>0 && dy>0)){
1648  x=x+1;
1649  }
1650  else{
1651  x=x-1;
1652  }
1653  py=py+2*(dx1-dy1);

1662 /**************************************************************************/
1680 /**************************************************************************/
1681 CRGB PixieChroma::kelvin_to_rgb(uint16_t temperature){
1682  // Tanner Helland formula
1683  float _temperature;
1684  float _red;
1685  float _green;
1686  float _blue;
1687 
1688  _temperature = constrain(temperature, 0, 65500);
1689 
1690  _red = _green = _blue = 0;
1691  float t = _temperature * 0.01;
1692 
1693  if (t <= 66){
1694  _red = 255;
1695  _green = (99.4708025861 * log(t)) - 161.1195681661;
1696  if (t > 19){
1697  _blue = (138.5177312231 * log(t - 10)) - 305.0447927307;
1698  }
1699  else{
1700  _blue = 0;
1701  }
1702  }
1703  else{
1704  _red = 329.698727466 * pow(t - 60, -0.1332047592);
1705  _green = 288.1221695283 * pow(t - 60, -0.0755148492);
1706  _blue = 255;
1707  }
1708 
1709  float f = 0.01 * 100;
1710 
1711  _red = constrain(f * _red, 0, 255);
1712  _green = constrain(f * _green, 0, 255);
1713  _blue = constrain(f * _blue, 0, 255);
1714 
1715  return CRGB(_red, _green, _blue);
1716 }
