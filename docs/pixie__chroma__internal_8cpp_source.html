39  extern PixieChroma pix;
40  // This is an imperfect solution, but extern-ing the class instance defined in the
41  // user sketch, this ISR routine (which can't be in a class) can still access it.
42  // _pixie_animations.cpp does this trick as well. This is why you cannot currently
43  // name your PixieChroma class instance anything but "pix". :/
44  //
45  // The solution to this would be the ISR setting a flag that a pix.process() function would read, but
46  // That would mean users can't ever delay() or use other blocking code for more than a split
47  // second without having to call the "pix.process()" again to avoid choppy output.
48  //
49  // At the cost of some non-standard methods of accessing the class in here, beginners to Arduino
50  // can just use simple print() and clear() methods to update the displays, while our
51  // tricky ISR keeps animation running nicely, even with blocking code in the main loop.

53  uint32_t t_now = micros(); // Get current time
54  pix.frame_time = (t_now - pix.t_last);

552 void PixieChroma::write_pix(char* message, int16_t x_pos, int16_t y_pos){
553  uint8_t len = strlen(message);
554  int16_t x_offset = x_pos;
555  int16_t y_offset = y_pos;
556  for(uint8_t i = 0; i < len; i++){
557  if(message[i] == '\n'){
558  y_offset += display_height;
559  x_offset = 1;
560  cursor_x = 1;
561  cursor_y += display_height;
562  }
563  else if(message[i] == 0 || message[i] == '\0'){ // early end of string
564  return;
565  }
566  else{
567  add_char(message[i], cursor_x, cursor_y);
568  x_offset += display_width;
569  cursor_x += display_width;
570  }
571  }
572 }

587  if (chr >= 32) {
588  chr -= 32;

595  uint16_t row2_index = xy(x_pos+x, y_pos+1);
596  uint16_t row3_index = xy(x_pos+x, y_pos+2);

605  mask[row3_index] = qadd8( mask[row3_index], bit_table[bitRead(column,2)] );

995 /**************************************************************************/
1001 /**************************************************************************/
1002 void PixieChroma::blur_x(fract8 blur_amount){
1003  uint8_t keep = 255 - blur_amount;
1004  uint8_t seep = blur_amount >> 1;
1005  for( uint8_t row = 0; row < matrix_height; row++) {
1006  uint8_t carryover = 0;
1007  for( uint8_t i = 0; i < matrix_width; i++) {
1008  uint8_t cur = mask[xy(i,row)];
1009  uint8_t part = cur;
1010  part = scale8(part, seep);
1011  cur = scale8(cur, keep);
1012  cur = qadd8(cur,carryover);
1013  if(i){
1014  mask[xy(i-1,row)] = qadd8(mask[xy(i-1,row)],part);
1015  }
1016  mask[xy(i,row)] = cur;
1017  carryover = part;
1018  }
1019  }
1020 }

1029 void PixieChroma::blur_y(fract8 blur_amount){
1030  // blur columns
1031  uint8_t keep = 255 - blur_amount;
1032  uint8_t seep = blur_amount >> 1;
1033  for( uint8_t col = 0; col < matrix_width; ++col) {
1034  uint8_t carryover = 0;
1035  for( uint8_t i = 0; i < matrix_height; ++i) {
1036  uint8_t cur = mask[xy(col,i)];
1037  uint8_t part = cur;
1038  part = scale8(part, seep);
1039  cur = scale8(cur, keep);
1040  cur = qadd8(cur,carryover);

1042  mask[xy(col,i-1)] = qadd8(mask[xy(col,i-1)],part);

1381  x=x1;

1395  px=px+2*dy1;
1396  }

1399  y=y+1;

1403  }

1463 CRGB PixieChroma::kelvin_to_rgb(uint16_t temperature){
1464  // Tanner Helland formula
1465  float _temperature;
1466  float _red;
1467  float _green;
1468  float _blue;
1469 
1470  _temperature = constrain(temperature, 0, 65500);
1471 
1472  _red = _green = _blue = 0;
1473  float t = _temperature * 0.01;
1474 
1475  if (t <= 66){
1476  _red = 255;
1477  _green = (99.4708025861 * log(t)) - 161.1195681661;
1478  if (t > 19){
1479  _blue = (138.5177312231 * log(t - 10)) - 305.0447927307;
1480  }
1481  else{
1482  _blue = 0;
1483  }
1484  }
1485  else{
1486  _red = 329.698727466 * pow(t - 60, -0.1332047592);
1487  _green = 288.1221695283 * pow(t - 60, -0.0755148492);
1488  _blue = 255;
1489  }
1490 
1491  float f = 0.01 * 100;
1492 
1493  _red = constrain(f * _red, 0, 255);
1494  _green = constrain(f * _green, 0, 255);
1495  _blue = constrain(f * _blue, 0, 255);
1496 
1497  return CRGB(_red, _green, _blue);
