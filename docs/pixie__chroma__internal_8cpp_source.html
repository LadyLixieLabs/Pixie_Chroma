1179 void PixieChroma::blur_x( fract8 blur_amount ){
1180  uint8_t keep = 255 - blur_amount;
1181  uint8_t seep = blur_amount >> 1;
1182  for( uint8_t row = 0; row < matrix_height; row++ ) {
1183  uint8_t carryover = 0;
1184  for( uint8_t i = 0; i < matrix_width; i++ ) {
1185  uint8_t cur = mask[xy( i,row )];
1186  uint8_t part = cur;
1187  part = scale8( part, seep );
1188  cur = scale8( cur, keep );
1189  cur = qadd8( cur,carryover );
1190  if( i ){
1191  mask[xy( i-1,row )] = qadd8( mask[xy( i-1,row )],part );
1192  }
1193  mask[xy( i,row )] = cur;
1194  carryover = part;
1195  }
1196  }
1197 }

1519 void PixieChroma::draw_line( int16_t x1, int16_t y1, int16_t x2, int16_t y2 ){

1521  uint16_t index;
1522 
1523  int16_t x,y,dx,dy,dx1,dy1,px,py,xe,ye,i;
1524  dx=x2-x1;

1527  dy1=fabs( dy );
1528  px=2*dy1-dx1;
1529  py=2*dx1-dy1;
1530  if( dy1<=dx1 ){
1531  if( dx>=0 ){
1532  x=x1;
1533  y=y1;
1534  xe=x2;
1535  }
1536  else{
1537  x=x2;
1538  y=y2;
1539  xe=x1;
1540  }
1541  index = xy( x,y );
1542  mask[index] = qadd8( 255, mask[index] );
1543  for( i=0;x<xe;i++ ){
1544  x=x+1;
1545  if( px<0 ){
1546  px=px+2*dy1;
1547  }
1548  else{
1549  if( ( dx<0 && dy<0 ) || ( dx>0 && dy>0 ) ){
1550  y=y+1;
1551  }
1552  else{
1553  y=y-1;
1554  }
1555  px=px+2*( dy1-dx1 );
1556  }
1557  delay( 0 );
1558  index = xy( x,y );
1559  mask[index] = qadd8( 255, mask[index] );
1560  }
1561  }
1562  else{
1563  if( dy>=0 ){
1564  x=x1;
1565  y=y1;
1566  ye=y2;
1567  }
1568  else{
1569  x=x2;

1614 CRGB PixieChroma::kelvin_to_rgb( uint16_t temperature ){
1615  // Tanner Helland formula
1616  float _temperature;
1617  float _red;
1618  float _green;
1619  float _blue;
1620 
1621  _temperature = constrain( temperature, 0, 65500 );
1622 
1623  _red = _green = _blue = 0;
1624  float t = _temperature * 0.01;
1625 
1626  if ( t <= 66 ){
1627  _red = 255;
1628  _green = ( 99.4708025861 * log( t ) ) - 161.1195681661;
1629  if ( t > 19 ){
1630  _blue = ( 138.5177312231 * log( t - 10 ) ) - 305.0447927307;
1631  }
1632  else{
1633  _blue = 0;
1634  }
1635  }
1636  else{
1637  _red = 329.698727466 * pow( t - 60, -0.1332047592 );
1638  _green = 288.1221695283 * pow( t - 60, -0.0755148492 );
1639  _blue = 255;
1640  }
1641 
1642  float f = 0.01 * 100;
1643 
1644  _red = constrain( f * _red, 0, 255 );
1645  _green = constrain( f * _green, 0, 255 );
1646  _blue = constrain( f * _blue, 0, 255 );
1647 
1648  return CRGB( _red, _green, _blue );
1649 }
