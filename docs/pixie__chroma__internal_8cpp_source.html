39  void ICACHE_RAM_ATTR ANIMATE(){
40  extern PixieChroma pix;
41  // This is an imperfect solution, but extern-ing the class instance defined in the
42  // user sketch, this ISR routine (which can't be in a class) can still access it.
43  // _pixie_animations.cpp does this trick as well. This is why you cannot currently
44  // name your PixieChroma class instance anything but "pix". :/
45  //
46  // The solution to this would be the ISR setting a flag that a pix.process() function would read, but
47  // That would mean users can't ever delay() or use other blocking code for more than a split
48  // second without having to call the "pix.process()" again to avoid choppy output.
49  //
50  // At the cost of some non-standard methods of accessing the class in here, beginners to Arduino
51  // can just use simple print() and clear() methods to update the displays, while our
52  // tricky ISR keeps animation running nicely, even with blocking code in the main loop.
53 
54  uint32_t t_now = micros(); // Get current time
55  pix.frame_time = (t_now - pix.t_last);

620 void PixieChroma::write_pix(char* message, int16_t x_pos, int16_t y_pos){
621  uint8_t len = strlen(message);
622  int16_t x_offset = x_pos;
623  int16_t y_offset = y_pos;
624  for(uint8_t i = 0; i < len; i++){
625  if(message[i] == '\n'){
626  y_offset += display_height;
627  x_offset = 1;
628  cursor_x = 1;
629  cursor_y += display_height;
630  }
631  else if(message[i] == 0 || message[i] == '\0'){ // early end of string
632  return;
633  }
634  else{
635  add_char(message[i], cursor_x, cursor_y);
636  x_offset += display_width;
637  cursor_x += display_width;
638  }
639  }
640 }

655 void PixieChroma::add_char(char chr, int16_t x_pos, int16_t y_pos){
656  if (chr >= 32) {
657  chr -= 32;
658  }
659 
660  for(uint8_t x = 0; x < 5; x++){
661  uint8_t column = pgm_read_byte(font+(chr * 5 + x));
662 
663  uint16_t row1_index = xy(x_pos+x, y_pos+0);
664  uint16_t row2_index = xy(x_pos+x, y_pos+1);
665  uint16_t row3_index = xy(x_pos+x, y_pos+2);
666  uint16_t row4_index = xy(x_pos+x, y_pos+3);
667  uint16_t row5_index = xy(x_pos+x, y_pos+4);
668  uint16_t row6_index = xy(x_pos+x, y_pos+5);
669  uint16_t row7_index = xy(x_pos+x, y_pos+6);
670 
671  // "Subtract" from mask transparency with saturating function
672  mask[row1_index] = qadd8( mask[row1_index], bit_table[bitRead(column,0)] );
673  mask[row2_index] = qadd8( mask[row2_index], bit_table[bitRead(column,1)] );
674  mask[row3_index] = qadd8( mask[row3_index], bit_table[bitRead(column,2)] );
675  mask[row4_index] = qadd8( mask[row4_index], bit_table[bitRead(column,3)] );
676  mask[row5_index] = qadd8( mask[row5_index], bit_table[bitRead(column,4)] );
677  mask[row6_index] = qadd8( mask[row6_index], bit_table[bitRead(column,5)] );
678  mask[row7_index] = qadd8( mask[row7_index], bit_table[bitRead(column,6)] );
679  }

1086 /**************************************************************************/
1092 /**************************************************************************/
1093 void PixieChroma::blur_x(fract8 blur_amount){
1094  uint8_t keep = 255 - blur_amount;
1095  uint8_t seep = blur_amount >> 1;
1096  for( uint8_t row = 0; row < matrix_height; row++) {
1097  uint8_t carryover = 0;
1098  for( uint8_t i = 0; i < matrix_width; i++) {
1099  uint8_t cur = mask[xy(i,row)];
1100  uint8_t part = cur;
1101  part = scale8(part, seep);
1102  cur = scale8(cur, keep);
1103  cur = qadd8(cur,carryover);
1104  if(i){
1105  mask[xy(i-1,row)] = qadd8(mask[xy(i-1,row)],part);
1106  }
1107  mask[xy(i,row)] = cur;
1108  carryover = part;
1109  }
1110  }
1111 }

1114 /**************************************************************************/
1120 /**************************************************************************/
1121 void PixieChroma::blur_y(fract8 blur_amount){
1122  // blur columns
1123  uint8_t keep = 255 - blur_amount;
1124  uint8_t seep = blur_amount >> 1;

1468 /**************************************************************************/
1477 /**************************************************************************/
1478 void PixieChroma::draw_line(int16_t x1, int16_t y1, int16_t x2, int16_t y2){
1479  //Bresenham's line algorithm
1480  uint16_t index;
1481 
1482  int16_t x,y,dx,dy,dx1,dy1,px,py,xe,ye,i;
1483  dx=x2-x1;
1484  dy=y2-y1;
1485  dx1=fabs(dx);
1486  dy1=fabs(dy);
1487  px=2*dy1-dx1;
1488  py=2*dx1-dy1;
1489  if(dy1<=dx1){
1490  if(dx>=0){
1491  x=x1;
1492  y=y1;
1493  xe=x2;
1494  }
1495  else{
1496  x=x2;
1497  y=y2;
1498  xe=x1;
1499  }
1500  index = xy(x,y);
1501  mask[index] = qadd8( 255, mask[index] );
1502  for(i=0;x<xe;i++){
1503  x=x+1;
1504  if(px<0){
1505  px=px+2*dy1;
1506  }
1507  else{
1508  if((dx<0 && dy<0) || (dx>0 && dy>0)){
1509  y=y+1;
1510  }
1511  else{
1512  y=y-1;
1513  }
1514  px=px+2*(dy1-dx1);
1515  }
1516  delay(0);
1517  index = xy(x,y);
1518  mask[index] = qadd8( 255, mask[index] );
1519  }
1520  }
1521  else{
1522  if(dy>=0){
1523  x=x1;
1524  y=y1;
1525  ye=y2;
1526  }
1527  else{
1528  x=x2;
1529  y=y2;
1530  ye=y1;
1531  }
1532  index = xy(x,y);
1533  mask[index] = qadd8( 255, mask[index] );
1534  for(i=0;y<ye;i++){
1535  y=y+1;
1536  if(py<=0){
1537  py=py+2*dx1;
1538  }
1539  else{
1540  if((dx<0 && dy<0) || (dx>0 && dy>0)){
1541  x=x+1;
1542  }
1543  else{
1544  x=x-1;
1545  }
1546  py=py+2*(dx1-dy1);
1547  }
1548  index = xy(x,y);
1549  mask[index] = qadd8( 255, mask[index] );
1550  }
1551  }
1552 }

1555 /**************************************************************************/
1573 /**************************************************************************/
1574 CRGB PixieChroma::kelvin_to_rgb(uint16_t temperature){
1575  // Tanner Helland formula
1576  float _temperature;
1577  float _red;
1578  float _green;
1579  float _blue;
1580 
1581  _temperature = constrain(temperature, 0, 65500);
1582 
1583  _red = _green = _blue = 0;
1584  float t = _temperature * 0.01;
1585 
1586  if (t <= 66){
1587  _red = 255;
1588  _green = (99.4708025861 * log(t)) - 161.1195681661;
1589  if (t > 19){
1590  _blue = (138.5177312231 * log(t - 10)) - 305.0447927307;
1591  }
1592  else{
1593  _blue = 0;
1594  }
1595  }
1596  else{
1597  _red = 329.698727466 * pow(t - 60, -0.1332047592);
1598  _green = 288.1221695283 * pow(t - 60, -0.0755148492);
1599  _blue = 255;
1600  }
1601 
1602  float f = 0.01 * 100;
1603 
1604  _red = constrain(f * _red, 0, 255);
1605  _green = constrain(f * _green, 0, 255);
1606  _blue = constrain(f * _blue, 0, 255);
1607 
1608  return CRGB(_red, _green, _blue);
1609 }
