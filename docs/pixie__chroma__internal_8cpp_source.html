39  void ICACHE_RAM_ATTR ANIMATE(){
40  extern PixieChroma pix;
41  // This is an imperfect solution, but extern-ing the class instance defined in the
42  // user sketch, this ISR routine (which can't be in a class) can still access it.
43  // _pixie_animations.cpp does this trick as well. This is why you cannot currently
44  // name your PixieChroma class instance anything but "pix". :/
45  //
46  // The solution to this would be the ISR setting a flag that a pix.process() function would read, but
47  // That would mean users can't ever delay() or use other blocking code for more than a split
48  // second without having to call the "pix.process()" again to avoid choppy output.
49  //
50  // At the cost of some non-standard methods of accessing the class in here, beginners to Arduino
51  // can just use simple print() and clear() methods to update the displays, while our
52  // tricky ISR keeps animation running nicely, even with blocking code in the main loop.
53 
54  uint32_t t_now = micros(); // Get current time
55  pix.frame_time = (t_now - pix.t_last);
56  pix.t_last = t_now;

729 void PixieChroma::write_pix(char* message, int16_t x_pos, int16_t y_pos){
730  uint8_t len = strlen(message);
731  int16_t x_offset = x_pos;
732  int16_t y_offset = y_pos;
733  for(uint8_t i = 0; i < len; i++){
734  if(message[i] == '\n'){
735  y_offset += display_height;
736  x_offset = 1;
737  cursor_x = 1;
738  cursor_y += display_height;
739  }
740  else if(message[i] == 0 || message[i] == '\0'){ // early end of string
741  return;
742  }
743  else{
744  add_char(message[i], cursor_x, cursor_y);
745  x_offset += display_width;
746  cursor_x += display_width;
747  }
748  }
749 }

764 void PixieChroma::add_char(char chr, int16_t x_pos, int16_t y_pos){
765  if (chr >= 32) {
766  chr -= 32;
767  }
768 
769  for(uint8_t x = 0; x < 5; x++){
770  uint8_t column = pgm_read_byte(font+(chr * 5 + x));
771 
772  uint16_t row1_index = xy(x_pos+x, y_pos+0);
773  uint16_t row2_index = xy(x_pos+x, y_pos+1);
774  uint16_t row3_index = xy(x_pos+x, y_pos+2);
775  uint16_t row4_index = xy(x_pos+x, y_pos+3);
776  uint16_t row5_index = xy(x_pos+x, y_pos+4);
777  uint16_t row6_index = xy(x_pos+x, y_pos+5);
778  uint16_t row7_index = xy(x_pos+x, y_pos+6);
779 
780  // "Subtract" from mask transparency with saturating function
781  mask[row1_index] = qadd8( mask[row1_index], bit_table[bitRead(column,0)] );
782  mask[row2_index] = qadd8( mask[row2_index], bit_table[bitRead(column,1)] );
783  mask[row3_index] = qadd8( mask[row3_index], bit_table[bitRead(column,2)] );
784  mask[row4_index] = qadd8( mask[row4_index], bit_table[bitRead(column,3)] );
785  mask[row5_index] = qadd8( mask[row5_index], bit_table[bitRead(column,4)] );
786  mask[row6_index] = qadd8( mask[row6_index], bit_table[bitRead(column,5)] );
787  mask[row7_index] = qadd8( mask[row7_index], bit_table[bitRead(column,6)] );
788  }

1195 /**************************************************************************/
1201 /**************************************************************************/
1202 void PixieChroma::blur_x(fract8 blur_amount){
1203  uint8_t keep = 255 - blur_amount;
1204  uint8_t seep = blur_amount >> 1;
1205  for( uint8_t row = 0; row < matrix_height; row++) {
1206  uint8_t carryover = 0;
1207  for( uint8_t i = 0; i < matrix_width; i++) {
1208  uint8_t cur = mask[xy(i,row)];
1209  uint8_t part = cur;
1210  part = scale8(part, seep);
1211  cur = scale8(cur, keep);
1212  cur = qadd8(cur,carryover);
1213  if(i){
1214  mask[xy(i-1,row)] = qadd8(mask[xy(i-1,row)],part);
1215  }
1216  mask[xy(i,row)] = cur;
1217  carryover = part;
1218  }
1219  }
1220 }

1577 /**************************************************************************/
1586 /**************************************************************************/
1587 void PixieChroma::draw_line(int16_t x1, int16_t y1, int16_t x2, int16_t y2){
1588  //Bresenham's line algorithm
1589  uint16_t index;
1590 
1591  int16_t x,y,dx,dy,dx1,dy1,px,py,xe,ye,i;
1592  dx=x2-x1;
1593  dy=y2-y1;
1594  dx1=fabs(dx);
1595  dy1=fabs(dy);
1596  px=2*dy1-dx1;
1597  py=2*dx1-dy1;
1598  if(dy1<=dx1){
1599  if(dx>=0){
1600  x=x1;
1601  y=y1;
1602  xe=x2;
1603  }
1604  else{
1605  x=x2;
1606  y=y2;
1607  xe=x1;
1608  }
1609  index = xy(x,y);
1610  mask[index] = qadd8( 255, mask[index] );
1611  for(i=0;x<xe;i++){
1612  x=x+1;
1613  if(px<0){
1614  px=px+2*dy1;
1615  }
1616  else{
1617  if((dx<0 && dy<0) || (dx>0 && dy>0)){
1618  y=y+1;
1619  }
1620  else{
1621  y=y-1;
1622  }
1623  px=px+2*(dy1-dx1);
1624  }
1625  delay(0);
1626  index = xy(x,y);
1627  mask[index] = qadd8( 255, mask[index] );
1628  }
1629  }
1630  else{
1631  if(dy>=0){
1632  x=x1;
1633  y=y1;
1634  ye=y2;
1635  }
1636  else{
1637  x=x2;
1638  y=y2;
1639  ye=y1;
1640  }
1641  index = xy(x,y);
1642  mask[index] = qadd8( 255, mask[index] );
1643  for(i=0;y<ye;i++){
1644  y=y+1;
1645  if(py<=0){
1646  py=py+2*dx1;
1647  }
1648  else{
1649  if((dx<0 && dy<0) || (dx>0 && dy>0)){
1650  x=x+1;
1651  }
1652  else{
1653  x=x-1;
1654  }
1655  py=py+2*(dx1-dy1);
1656  }
1657  index = xy(x,y);
1658  mask[index] = qadd8( 255, mask[index] );
1659  }
1660  }
1661 }

1664 /**************************************************************************/
1682 /**************************************************************************/
1683 CRGB PixieChroma::kelvin_to_rgb(uint16_t temperature){
1684  // Tanner Helland formula
1685  float _temperature;
1686  float _red;
1687  float _green;
1688  float _blue;
1689 
1690  _temperature = constrain(temperature, 0, 65500);
1691 
1692  _red = _green = _blue = 0;
1693  float t = _temperature * 0.01;
1694 
1695  if (t <= 66){
1696  _red = 255;
1697  _green = (99.4708025861 * log(t)) - 161.1195681661;
1698  if (t > 19){
1699  _blue = (138.5177312231 * log(t - 10)) - 305.0447927307;
1700  }
1701  else{
1702  _blue = 0;
1703  }
1704  }
1705  else{
1706  _red = 329.698727466 * pow(t - 60, -0.1332047592);
1707  _green = 288.1221695283 * pow(t - 60, -0.0755148492);
1708  _blue = 255;
1709  }
1710 
1711  float f = 0.01 * 100;
1712 
1713  _red = constrain(f * _red, 0, 255);
1714  _green = constrain(f * _green, 0, 255);
1715  _blue = constrain(f * _blue, 0, 255);
1716 
1717  return CRGB(_red, _green, _blue);
1718 }
