1221 void PixieChroma::blur_x( fract8 blur_amount ){
1222  uint8_t keep = 255 - blur_amount;
1223  uint8_t seep = blur_amount >> 1;
1224  for( uint8_t row = 0; row < matrix_height; row++ ) {
1225  uint8_t carryover = 0;
1226  for( uint8_t i = 0; i < matrix_width; i++ ) {
1227  uint8_t cur = mask[xy( i,row )];
1228  uint8_t part = cur;
1229  part = scale8( part, seep );
1230  cur = scale8( cur, keep );
1231  cur = qadd8( cur,carryover );
1232  if( i ){
1233  mask[xy( i-1,row )] = qadd8( mask[xy( i-1,row )],part );
1234  }
1235  mask[xy( i,row )] = cur;
1236  carryover = part;
1237  }
1238  }
1239 }

1586 void PixieChroma::draw_line( int16_t x1, int16_t y1, int16_t x2, int16_t y2 ){

1588  uint16_t index;
1589 
1590  int16_t x,y,dx,dy,dx1,dy1,px,py,xe,ye,i;
1591  dx=x2-x1;
1592  dy=y2-y1;
1593  dx1=fabs( dx );
1594  dy1=fabs( dy );
1595  px=2*dy1-dx1;
1596  py=2*dx1-dy1;
1597  if( dy1<=dx1 ){
1598  if( dx>=0 ){
1599  x=x1;
1600  y=y1;
1601  xe=x2;
1602  }
1603  else{
1604  x=x2;
1605  y=y2;
1606  xe=x1;
1607  }

1610  for( i=0;x<xe;i++ ){
1611  x=x+1;
1612  if( px<0 ){
1613  px=px+2*dy1;
1614  }
1615  else{
1616  if( ( dx<0 && dy<0 ) || ( dx>0 && dy>0 ) ){
1617  y=y+1;
1618  }
1619  else{
1620  y=y-1;
1621  }
1622  px=px+2*( dy1-dx1 );
1623  }
1624  delay( 0 );
1625  index = xy( x,y );
1626  mask[index] = qadd8( 255, mask[index] );
1627  }
1628  }
1629  else{
1630  if( dy>=0 ){
1631  x=x1;
1632  y=y1;
1633  ye=y2;
1634  }
1635  else{
1636  x=x2;
1637  y=y2;
1638  ye=y1;
1639  }
1640  index = xy( x,y );
1641  mask[index] = qadd8( 255, mask[index] );

1645  py=py+2*dx1;
1646  }
1647  else{

1652  x=x-1;
1653  }

1683 CRGB PixieChroma::kelvin_to_rgb( uint16_t temperature ){

1700  }
1701  else{

1704  }
